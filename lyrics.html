Define your list of colors here const colors = ["#FF5733", "#33FF57", "#3357FF", "#F3FF33", "#FF33F3", "#33FFF0"]; // Function to get a random color that is different from the last one function getRandomColor() { // Retrieve the last used color from localStorage const lastColor = localStorage.getItem('lastPageColor'); // Filter out the last color to ensure no immediate repetition const availableColors = colors.filter(color => color !== lastColor); // If for some reason no colors are left (e.g., list had only 1 color), fallback to the full list const colorPool = availableColors.length > 0 ? availableColors : colors; // Pick a random color from the pool const newColor = colorPool[Math.floor(Math.random() * colorPool.length)]; // Save the new color to localStorage for the next refresh/change localStorage.setItem('lastPageColor', newColor); return newColor; } // Function to apply the color change (e.g., changing the background) function applyColorChange() { const chosenColor = getRandomColor(); document.body.style.backgroundColor = chosenColor; // Changes the body background console.log(`Color changed to: ${chosenColor}`); } // Function to calculate the exact milliseconds until the next hour or half-hour function getMsUntilNextInterval() { const now = new Date(); const minutes = now.getMinutes(); const seconds = now.getSeconds(); const milliseconds = now.getMilliseconds(); let minutesRemaining; if (minutes < 30) { // Time remaining until the half-hour (:30) minutesRemaining = 30 - minutes; } else { // Time remaining until the top of the next hour (:00) minutesRemaining = 60 - minutes; } // Convert everything to milliseconds return (minutesRemaining * 60 * 1000) - (seconds * 1000) - milliseconds; } // Function to manage the clock synchronization function startHourlyHalfHourlyTimer() { const msToNextInterval = getMsUntilNextInterval(); // Set a one-time timeout to trigger exactly at the next :00 or :30 mark setTimeout(() => { applyColorChange(); // Change color on the interval // Once synchronized, trigger every 30 minutes (30 mins * 60 secs * 1000 ms = 1800000 ms) setInterval(applyColorChange, 1800000); }, msToNextInterval); } // --- INITIALIZATION ON PAGE LOAD --- // 1. Change the color immediately when the page refreshes/loads applyColorChange(); // 2. Start the tracking timer to catch the next hour or half-hour mark startHourlyHalfHourlyTimer();
